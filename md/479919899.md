/**
 * 求次方, base 的 exponent 次方
 * @param {*} base 
 * @param {*} exponent 
 */
var seekPower = (base, exponent) => {
  if (exponent === 0) return 1;
  return base * seekPower(base, exponent - 1);
};

/**
 * 开根号
 * @param {*} base 
 * @param {*} exponent 
 */
var seekRoot = (base, exponent) => {
  // base大于0，开根号
  if (base > 0) {
    let newExponent = 1 / exponent;
    
    // 猜测值。如果大了就减去item，如果小了就把之前的加回去，让item/10
    let newBase = base;
    let item = 1;
    while (_isEqual(base, newBase, newExponent)) {
      if (base < seekPower(newBase, newExponent)) {
        newBase -= item;
      }
      if (base > seekPower(newBase, newExponent)) {
        newBase += item;
        item = item / 10;
      }
    }
    return newBase;
  }
  // base等于0，返回0
  if (base === 0) {
    return 0;
  }
  // base小于0，返回NaN
  if (base < 0) {
    return NaN;
  }

  function _isEqual (base, newBase, newExponent) {
    // 精确到小数点后14位
    var coefficient = 100000000000000
    return (
      Math.floor(base * coefficient) !==
      Math.floor(seekPower(newBase, newExponent) * coefficient)
    );
  };
};

/**
 * 求幂
 * @param {*} base 底数
 * @param {*} exponent 指数
 */
var MathPow = (base, exponent) => {
  // 如果指数大于1， 求次方
  if (exponent > 1) {
    return seekPower(base, exponent);
  }
  // 如果指数在0和1之前，开根号
  if (1 > exponent && exponent > 0) {
    return seekRoot(base, exponent);
  }
  // 如果指数在0和1之前，开根号
  if (exponent === 0) {
    return 1;
  }
  // 如果指数为1，直接返回底数
  if (exponent === 0) {
    return base;
  }

  return NaN;
};

Math.pow()参数包含了以下几种输入：
1. Math.pow(2,2), 普通的幂运算
2. Math.pow(2, -2), 幂运算的倒数运算
3. Math.pow(4, 1/2),Math.pow(4, 1/3)。对4开二次方，开三次方。
所以，需要包含很多种情况：


```javascript
var myPow = function(x, n) {
    if (n === 0) {
         return 1;
    }
    if (n === 1) {
        return x;
    }
    if (n <= -1) {
      // 指数为负数，则求幂的倒数
        return 1 / myPow(x, -n);
    } else if (n > 1) {
        return x * myPow(x, n-1)
    } else {
     //开方运算
        return sqrt(x, 1/n)
    }
}
var sqrt = function(x, n) {
    if (!n) n = 2;
    if (n < 0) {
        return 1 / sqrt(x, -n)
    }
    var left = 0;
    var right = x;
    var mid = left + (right-left) / 2;
    while (true) {
        if (Math.abs(myPow(mid, n)) === x) {
            return mid;
        };
         if(mid > x / mid) {
                right = mid;
            } else {
                left = mid;
            }
         mid = left + (right - left) / 2;
    }
    return mid;
}
// myPow(2,2) => 4
// myPow(2,-2) => 0.25
// myPow(2,1/2) => 1.414
// myPow(8, 1/3) => 2
// myPow(8,-1/3) => 0.5

```
