```javascript
/**
 * @param {number} n
 * @param {number} k
 * @return {number[][]}
 */

var combine = function(n, k) {
    var res = []
    
    function step(start, item) {
        if (item.length === k) {
            res.push([...item])
            return
        }
        for (let i = start; i < n+1; ++i) {
            // 加入第一个数
            item.push(i)
            // 加入第二个数
            step(i+1, item)
            // 回退
            item.pop()
        }
    }
    
    step(1, [])
    
    return res
};

```
